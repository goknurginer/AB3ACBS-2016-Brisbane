\begin{frame}
\frametitle{Downsides of Roast}
Limitations when analysing large collections of sets\\
\vfill
 \textbf{\color{oxygenpurple} P-value resolution is limited by the number of rotations}\\
\vfill
Top detected sets are all have the same p-values \\
\vfill
P-values for each set may vary from run to run\\
\vfill
\includegraphics[width=1\textwidth]{figures/roast_table}
\end{frame}

\begin{frame}
\frametitle{Improved Features with Fry}
Assumes equal gene-wise variances\\
\vfill
 \textbf{\color{oxygenpurple} Produces high-resolution exact p-values unlike roast}\\
\vfill
Therefore gene level statistics are standardized using posterior gene variances \\
\vfill
Computationally very efficient\\
\vfill
\includegraphics[width=1\textwidth]{figures/fry_table}
\end{frame}
